\begin{definition}[Heat Equation and Adjoint equation]
    The heat equation
    \[
    \partial_t u = \Delta u,\quad u\in \mathbb{R}^n\times\mathbb{R}
    \]
    has a class of solution
    \[
    \Gamma(x,t;\xi,\tau) = \dfrac{1}{(2\sqrt{\pi})^n}(t-\tau)^{-n/2}\exp\left(-\dfrac{|x-\xi|^2}{4(t-\tau)}\right),\quad (\xi,\tau) \in \mathbb{R}^n\times\mathbb{R}
    \]
    where for any continuous function $f(x) \lesssim \expo{h|x|^2}$ for some $h>0$, then the integral
    \[u(x,t) = \int \Gamma(x,t;\xi,0)f(\xi)d\xi\]
    exists for $0<t\leq T$ if $T<1/4h$ and $u(x,t) \to f(x), t\to 0$, and $\Gamma$ is called a fundamental solution of the heat equation; and $u$ is a solution of the Cauchy problem, i.e. the problem of finding a solution to the heat equation for $0<t \leq T$ with the initial condition $u(x,0) = f(x)$.\par
    The adjoint equation if defined as
    \[
    \partial_t v = - \Delta v,\quad v\in \mathbb{R}^n\times\mathbb{R}
    \]
    and in this case $\Gamma(x,t;\xi,\tau)$ is a solution for the adjoint equation as a function of $(\xi,\tau)$ with fixed $(x,t)$.
\end{definition}


\begin{definition}[Holder Space]
    
A function $f(x)$ defined on a bounded    closed set $S$ of $\R^n$ is said to be Holder continuous of exponent $\alpha(0 < \alpha < 1)$ in $S$ if there exists a constant $A$ such that \[|f(x)-f(y)| \leq A|x-y|^{\alpha}\] for all $x, y$ in $S$. The smallest $A$ for the inequality is called the Holder coefficient.\par

If $S$ is an unbounded set whose intersection with every bounded closed set $V$ is closed, then $f(x)$ is said to be Holder continuous of exponent $a$ in $S$ if for every bounded closed set $V$ the equality holds in $S \cap V$ with some $A$ which may depend on $V$. If $A$ can be taken independently of $V$, then we say that $f(x)$ is uniformly  Holder continuous.\par
If $S$ is an open set then $f(x)$ is said to be locally Holder continuous (exponent  $\alpha$) in $S$ if the inequality holds in every bounded closed set $V \subset S$, with
$A$ which may depend on $V$. If $A$ is independent of $V$ then $f(x)$ is uniformly Holder continuous (exponent $\alpha$).\par
If $f$ depends on a parameter $\lambda$, i.e. $f = f(x,\lambda)$, and if the Holder
coefficient is independent of $\lambda$, then we say that $f(x,\lambda)$ is Holder continuous in $x$, uniformly with respect to $\lambda$.\par
If in the inequality $\alpha = 1$ then we say that $f(x)$ is Lipschitz continuous.
\end{definition}

\begin{definition}[Parabolic Equation]

Consider the differential equation
\[
    Lu \equiv \sum\limits_{i,j=1}^n a_{ij}(x,t) \partial_{x_i}\partial_{x_j} u + \sum\limits_{i=1}^n b_i(x,t)\partial_{x_i} u + c(x,t)u - \partial_t u =0
\]
where the coefficients $a_{ij}, b_i, c$ are defined in a cylinder
\[
\Omega \equiv \overline{D}\times [T_0,T_1] \equiv \{(x,t);x\in \overline{D}, T_0 \leq t \leq T_1\}
\]
$\overline{D}$ is the closure of a bounded domain $D\subset\R^n$. We always take $a_{ij}(x,t)$ to be a symmetric matrix. If the matrix is positive
definite, then we say that the operator $L$ is of parabolic type at the point $(x, t)$. If $L$ is parabolic at all the points of $\Omega$ then we say that $L$ is parabolic in $\Omega$. If there exist positive constants $\overline{\lambda}_0, \overline{\lambda}_1$ such that, for any real vector $\xi$,
\[\overline{\lambda}_0|\xi|^2 \leq \sum\limits_{i,j=1}^n a_{ij}(x,t)\xi_i\xi_j \leq \overline{\lambda}_1|\xi|^2\]
for all $(x,t) \in \Omega$ then we say that $L$ is uniformly parabolic in $\Omega$.
\end{definition}

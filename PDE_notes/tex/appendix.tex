\section{Functional Analysis}

\subsection{Banach Spaces}

\begin{theorem}[Hahn-Banach Theorem]\ \par
    Let $\mathcal{X}$ be a real vector space, $p$ a sublinear functional, i.e.
    \[
    p(x+y) \leq p(x)+p(y),\quad p(\lambda x) = \lambda p(x)\text{ for any }x,y\in\mathcal{X}\text{ and }\lambda \geq 0
    \]
    on $\mathcal{X}$, $\mathcal{M}$ a subspace of $\mathcal{X}$, and $f$ a linear functional on $\mathcal{M}$ such that $f(x) \leq p(x)$ for all $x\in\mathcal{M}$. Then there exists a linear functional $F$ on $\mathcal{X}$ such that $F(x) \leq p(x)$ for all $x\in \mathcal{X}$ and $F|_{\mathcal{M}} = f$.
\end{theorem}

\begin{proposition}
    If $f\in C^{\infty}$, then $f\in \Schw$ iff $x^{\beta}\partial^{\alpha}f$ is bounded for all multi-indices $\alpha,\beta$ iff $\partial^{\alpha}( x^{\beta}f)$ is bounded for all multi-indices $\alpha,\beta$.
\end{proposition}
\begin{proof}
    (1) $\implies$ (2) $\Leftrightarrow$ (3) is quite simple. For (2) $\implies$ (1), here is a good way to show: for any integer $N$, we know that $\sum\limits_{i=1}^n|x_i|^N$ has a positive minimum $\delta$ on the unit sphere. So we will know that $\sum\limits_{i=1}^n |x_i|^N \geq \delta|x|^N$ and then
    \[
    (1+|x|)^N \leq 2^N(1+|x|^N) \leq 2^N(1+\delta^{-1}\sum\limits_{i=1}^n |x_j^N|) \leq 2^N\delta^{-1}\sum\limits_{|\beta| \leq N}|x^{\beta}|
    \]
\end{proof}

\begin{definition}[Shift Operator]\ \par
    If $f$ is a function on $\R^m$ and $y\in \R^m$,
    \[\tau_y f(x) = f(x-y)\]
    $f$ is called uniformly continuous if $||\tau_y f - f||_u \to 0$ as $y$.
\end{definition}

\begin{lemma}
    If $f\in C_c(\R^n)$, then $f$ is uniformly continuous.
\end{lemma}

\begin{proposition}
    If $1\leq p <\infty$, translation is continuous in the $L^p$ norm, i.e. if $f\in L^p$ and $z\in \mathbb{R}^n$, then $\lim_{y\to 0} ||\tau_{y+z}f - \tau_z(f)||_p = 0$.
\end{proposition}
\begin{proof}
    Notice for some $g\in C_c(\R^n)$, we have
    \[
    ||\tau_{y+z}f-\tau_zf||_p \leq ||\tau_{y+z} f - \tau_{y+z}g||_p + ||\tau_{y+z}g-\tau_z g||_p + ||\tau_z f- \tau_z g||_p
    \]
    and choose $g_{\epsilon}$ to be some function that $||g_{\epsilon}-f||_p < \epsilon$, so
    \[
    \limsup_{y\to 0} ||\tau_{y+z}f-\tau_zf||_p \leq ||\tau_{y+z} f - \tau_{y+z}g||_p + \limsup_{y\to 0} ||\tau_{y+z}g-\tau_z g||_p + ||\tau_z f- \tau_z g||_p < 2\epsilon
    \]
    and we are done.
\end{proof}

\begin{definition}[Convolution]\ \par
    For $f,g$ measurable on $\R^n$, the convolution $f*g$ is defined by
    \[
    f*g(x) = \int f(x-y)g(y)dy
    \]
\end{definition}

\begin{proposition}[Young's Inequality]\ \par
    If $f\in L^1$ and $g\in L^p(1\leq p \leq \infty)$, then $f*g(x)$ exists for almost every $x$, $f*g \in L^p$ and $||f*g||_p \leq ||f||_1||g||_p$.
\end{proposition}

\section{Sobolev Spaces}

\subsection{Sobolev Spaces}

\begin{definition}
    Denote $\Tst(U)$ to be the test functions. Let $u \in \Tst'(U )$, where $U \subset \mathbb{R}^d$ is open. The $k$-th order $L^p$-based Sobolev norm of $u$ is
    \[||u||_{W^{k,p}(U)}: = \sum_{|\alpha|\leq k} ||D^{\alpha}u||_{L^p},\quad 1 \leq p < \infty\]
    where we are using the distributional derivative and assume that $D^{\alpha}u$ is an $L^p$ function.
\end{definition}

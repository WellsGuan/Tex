\part{Sobolev Spaces}

In the following chapters, we denote $\D(U)$ to be the test functions on an open set $U\subset\mathbb{R}^m$, i.e. the continuous functions with compact support on $U$, and $V'$ to be the dual space of the vector space $V$.

\section{Introduction to Sobolev Spaces}
\subsection{Sobolev spaces}
\begin{definition}
Let $u\in \D^{\prime}(U)$ , where \(U\subset \mathbb{R}^{d}\) is open. The \(k\)-th order \(L^{p}\)-\emph{based Sobolev norm} of \(u\) is
\[
\| u\|_{W^{k,p}(U)} := \sum_{|\alpha |\leq k} \| D^{\alpha} u\|_{L^{p}}, \quad 1 \leq p < \infty ,
\]
where we are using the distributional derivative and assume that \(D^{\alpha} u\) is an \(L^{p}\) function.
\end{definition}

\begin{definition}
Let \(U\) be an open subset of \(\mathbb{R}^{d}\). The \(L^{p}\)-\emph{based Sobolev space} of order \(k\) on \(U\) is
\[
W^{k,p}(U) = \{u\in D'(U):\| u\|_{W^{k,p}(U)}< \infty \} .
\]
Note that \(\D(U) \subseteq W^{k,p}(U)\).
\end{definition}

\begin{definition}
The set of \(u \in W^{k,p}(U)\) that vanish (to appropriate orders) on \(\partial U\) is
\[
W_{0}^{k,p}(U) = \overline{C_{c}^{\infty}(U)}^{\| \cdot \|}_{W^{k,p}}\subset W^{k,p}(U).
\]
When \(p = 2\) , we introduce the notation
\[
H^{k}(U) = W^{k,2}(U), \qquad H_{0}^{k}(U) = W_{0}^{k,2}(U).
\]
We can define an inner product on \(H^{k}\) by
\[
\langle u,v\rangle_{H^{k}} = \sum_{|\alpha |\leq k}\langle D^{\alpha}u,D^{\alpha}v\rangle_{L^{2}}.
\]
\end{definition}

\begin{proposition}\ \par
\begin{enumerate}
    \item For all \(k \in \mathbb{Z}_{\geq 0}\) and \(1 \leq p \leq \infty\) , the following Sobolev spaces \((W^{k,p}(U), \| \cdot \|_{W^{k,p}})\) and \((W_{0}^{k,p}(U), \| \cdot \|_{W^{k,p}})\) are Banach spaces.
    \item For all \(k \in \mathbb{Z}_{\geq 0}\) , \((H^{k}(U), \langle \cdot , \cdot \rangle_{H^{k}})\) and \((H_{0}^{k}(U), \langle \cdot , \cdot \rangle_{H^{k}})\) are Hilbert spaces.
    \item (Fourier-analytic characterization of \(H^{k}\) ) Given \(u \in H^{k}(U)\)
\[
\| u\|_{H^{k}} \simeq \| \hat{u} \|_{L^{2}} + \| \xi |^{k} \hat{u} \|_{L^{2}} \simeq \| (1 + |\xi |^{2})^{k / 2} \hat{u} \|_{L^{2}},
\]
where $\simeq$ between norms are the equivalence.
\end{enumerate}
\end{proposition}

\subsection{Negative Sobolev Spaces}

It is natural to consider the dual space of the Sobolev spaces, which can be interpreted as the following space as well.

\begin{definition}
    For $k>0$ and $1<p<\infty$, define the \emph{negative order Sobolev norm}
\[
\|u\|_{W^{-k,p}(U)}
=
\inf
\left\{
\sum_{|\alpha|\le k}
\|g_\alpha\|_{L^p}
:
u=\sum_{|\alpha|\le k} D^\alpha g_\alpha
\right\}.
\]
and the \emph{negative order Sobolev space} is
\[
W^{-k,p}(U)
=
\left\{
u\in\mathcal{D}'(U):
u=\sum_{|\alpha|\le k} D^\alpha g_\alpha,
\ g_\alpha\in L^p(U)
\right\}.
\]
\end{definition}

\begin{proposition}
For $k\in\mathbb{Z}_{\ge0}$ and $1<p<\infty$,
\[
\bigl(W^{k,p}_0(U)\bigr)^*
= W^{-k,q}(U),
\qquad
\frac1p+\frac1q=1.
\]
\end{proposition}
\begin{proof}
For $u\in \D(U)$ and some
\[
v = \sum\limits_{|\alpha| \leq k} D^{\alpha}g_{\alpha},\quad  g_{\alpha} \in L^q(U),
\]
and we may define $(\cdot)^*: W^{-k,q}(U) \to (W^{k,p}_0(U))^*$ by
\[
v^{\flat}(u) = \int vu = \sum\limits_{|\alpha| \leq k} (-1)^{|\alpha|}\int g_{\alpha} D^{\alpha}u
\]
which is well-defined since another representation is equal to this from the distributions' perspectives on $\D(U)$, where the equality can be extended to the whole $W^{k,p}_0(U)$ by the density argument that
\[
|v^{\flat}(u)| \leq \sum\limits_{|\alpha| \leq k}\left|\int g_{\alpha} u\right| \leq \sum\limits_{|\alpha| \leq k}||g_{\alpha}||_{L^q}||D^{\alpha}u||_{L^p} \leq 2||v||_{W^{-k,q}(U)} ||u||_{W^{k,p}(U)} 
\]
where we can choose the representation of $v$ to make
\[
\sum\limits_{|\alpha| \leq k}||g_{\alpha}||_{L^p} \leq 2||v||_{W^{-k,q}(U)}
\]
and this help us to make the extension.
\par
From the other side, we need to show for any linear functional on $W^{k,p}_0(U)$ can be obtained by some element in the negative Sobolev space. Firstly, notice there is a natural inclusion of $W_0^{k,p}(U)$ to $L^p(U)^{\oplus K(k)}$, and define a linear map
\[
T:\D(U) \to L^p(U)^{\otimes K(k)}, \quad u\mapsto (u,D_{x_1}u,\cdots, D_{x^{k}}u,\cdots,D^{\alpha}u)
\]
and then we may see that $T$ is an isometry between \[(\D(U),||\cdot||_{W^{k,p}}) \text{ and } (T(\D(U)),|\cdot|_{L^p(U)^{\otimes K(k)}}).\]
For a linear functional $l$ on $W^{k,p}_p(U)$, we may know $l$ will induce a bounded linear functional
\[
\tilde{l}: T(\D(U)) \to\mathbb{R},\quad \tilde{l}(Tu) = l(u)
\]
which can be extended to the whole $L^p(U)^{\otimes K(k)}$ by Hahn-Banach and denote the extended linear functional to be $l'$.

\end{proof}

\subsection{Relationship between a priori estimates, existence, and uniqueness}

\begin{proposition}
    Let $X,Y$ be Banach spaces, and $P:X\to Y \label{dop}$ be bounded linear operator. Denote $P^*:Y^*\to X^*$ as the adjoint, i.e. $\langle v,Pu\rangle
= \langle P^*v,u\rangle.$ Assume for any $u\in X$ we have $\|u\|_X\le C\|Pu\|_Y$, then
\begin{enumerate}
\item If $Pu=0$, then $u=0$.
\item For all $g\in X^*$,
there exists $v\in Y^*$
such that
\[
P^*v=g,
\qquad
\|v\|_{Y^*}\le C\|g\|_{X^*}.
\]
\end{enumerate}
\end{proposition}

\begin{proof}
    (1) is obvious. For (2), consider $PX:=\{y\in Y,\exists\  x\in X\text{ s.t.} Px=y\}$ which is a subspace of $Y$, and define $\bar{v}(Px) = \langle g,x\rangle$ which is a well-defined linear functional on $PX$ with
    \[
    ||\bar{v}(Px)|| \leq ||g||_{X^*}||x||_X \leq C||g||_{X^*}||Px||_Y
    \]
    and hence we know there is a linear functional $v \in Y^*$ such that $v|_{PX} = \bar{v}$ and $||v||_{Y^*} \leq C||g||_{X^*}$ and we are done.
\end{proof}


\begin{definition}
    Let $X$ be a Banach space, we call it \emph{reflexive} if $u\to(v\mapsto\langle v,u \rangle)$ is an isomorphism.
\end{definition}

\begin{proposition}
    Assume $X$ is reflexive. Let $X,Y$ be Banach spaces, and let $P:X\to Y$ be bounded linear operator. Suppose $||v||_{Y^{*}} \leq C||P^*v||_{X^*}$. Then
    \begin{enumerate}
        \item If $v \in Y^*$ and $P^*v = 0$, then $v = 0$.
        \item For all $f\in Y$, there exists a $u\in X$ such that $Pu = f$ and $||u||_X \leq C||f||_Y$.
    \end{enumerate}
\end{proposition}

\emph{Remark.} All sobolev spaces $W^{k,p}_0(U)$ with $1 < p < \infty$ are reflexive.

\begin{definition}
    Given $U\subset Y$, the \emph{annihilator} of $U$ is
    \[
    U^{\perp}:=\{v\in Y^*:\langle v,f\rangle = 0,\text{ for any }f\in U\}
    \]
    and similarly, for some $V\subset X^*$, the \emph{annihilator} of $V$ is
    \[
    ^{\perp}V:=\{u\in X:\langle g,u\rangle = 0,\text{ for any }g\in V\}
    \]
\end{definition}

\begin{proposition}
    For the operators in $\eqref{dop}$ we have 
    \[
    (\ran{P})^{\perp} = \ker(P^*),\quad \ker(P) = ^\perp(\ran{P^*})
    \]
\end{proposition}
\begin{proof}
    For the second equality, refer to Hahn-Banach to show that if some $u\in X$ is in the kernel of any linear functional in $X^*$, then $u = 0$.
\end{proof}

\begin{proposition}
    Let $X,Y$ be Banach spaces, with $X$ reflexive, and $P:X\to Y$ be a bounded linear operator. If $P(X) = Y$, then thre exists some $C>0$ such that
    \[||v||_{Y^*} \leq C||P^*v||_{X^*}\]
\end{proposition}

\begin{example}
    Try to show the $1$-dimensional Laplace equation
    \[
    \begin{cases}
        -u'' = f\quad &\text{in }(0,1), \\
        u=0&\text{at }x=0,1
    \end{cases}
    \]
    will have a unqiue solution in $H^1(0,1)$.
\end{example}

\section{Trace and Extension Theorems and Introduction to Sobolev Inequalities}
\subsection{The trace theorem}
\begin{definition}
For \(u\in C^{1}(\overline{U})\) , we define the trace to be
\[
\mathrm{tr}_{\partial U}u:= u|_{\partial U}.
\]
We wish to extend this operation to all of \(W^{1,p}(U)\) . Note that \(\mathrm{tr}_{\partial U}\) is linear, so we can extend it if we know it is bounded.
\end{definition}

\begin{theorem}[Trace theorem, non- sharp]
Let \(U\) be a bounded open subset of \(\mathbb{R}^{d}\) with \(C^{1}\) boundary \(\partial U\) , and let \(1< p< \infty\) . Then for \(u\in C^{1}(\overline{U})\)
\[
\| \mathrm{tr}_{\partial U}u\|_{L^p (\partial U)}\leq C\| u\|_{W^{1,p}(U)}.
\]
(i) As a consequence, \(\mathrm{tr}_{\partial U}\) extends uniquely by continuity (and density of \(C^{1}(\overline{U})\subset W^{1,p}(U)\) ) to a bounded linear map
\[
\mathrm{tr}_{\partial U}:W^{1,p}(U)\to L^p (\partial U).
\]
(ii) Moreover, \(u\in W_{0}^{1,p}(U)\) if and only if \(\mathrm{tr}_{\partial U}u = 0\).
\end{theorem}

\begin{remark}
    The map \(\mathrm{tr}_{\partial U}:W^{1,p}(U)\to L^{p}(\partial U)\) is not surjective.
\end{remark}

\begin{definition}
    The \emph{fractional ($L^2$-based) Sobolev spaces} is defined by the set of $L^2$ function $u$ with

\end{definition}

\begin{theorem}
There exists a bounded linear map
\[
\operatorname{ext}_{\partial U}: H^{1/2}(\mathbb{R}^{d-1}) \to H^1(\mathbb{R}_+^d)
\]
such that
\[
\operatorname{tr}_{\partial U}\circ \operatorname{ext}_{\partial U} = id.
\]
\end{theorem}

\begin{remark}
This theorem asserts that every boundary function with the optimal trace regularity can be realized as the boundary value of an interior $H^1$ function.
\end{remark}

\begin{proof}
We use the Poisson semigroup. Given $g\in \mathcal{S}(\mathbb{R}^{d-1})$, define its Poisson extension $v$ by
\[
v(x',x_d) = (P_{x_d} * g)(x'), \qquad \widehat{P_{x_d}}(\xi') = e^{-x_d|\xi'|}.
\]
Equivalently, $v$ is the unique solution to the Dirichlet problem
\[
\begin{cases}
\Delta v = 0 & \text{in }\mathbb{R}_+^d,\\
v(x',0)=g(x') & \text{on }\partial \mathbb{R}_+^d.
\end{cases}
\]
Choose $\eta\in C_c^{\infty}(\mathbb{R})$ such that $\eta=1$ for $|s|<1$ and $\eta=0$ for $|s|>2$, and define
\[
u = \operatorname{ext}_{\partial U}g := \eta(x_d)v.
\]
Then
\[
\mathcal{F}_{x'}u(\xi',x_d) = \eta(x_d)e^{-x_d|\xi'|}\hat{g}(\xi').
\]
One checks that $u\in H^1(\mathbb{R}_+^d)$ and that
\[
\|u\|_{H^1(\mathbb{R}_+^d)} \lesssim \|g\|_{H^{1/2}(\mathbb{R}^{d-1})}.
\]
Moreover, since $\eta(0)=1$, we have
\[
\operatorname{tr}_{\partial U}u = g.
\]
Therefore this construction gives a bounded right inverse to the trace map.
\end{proof}

\end{remark}

Thus far we have described Sobolev regularity in terms of weak derivatives and $L^p$ norms. There is another complementary point of view, especially natural in the $L^2$ setting, based on the Fourier transform. Roughly speaking, irregular behavior corresponds to large oscillation at small scales, and the Fourier transform is designed to isolate such oscillations.

For $f\in L^1(\mathbb{R}^d)$, define the Fourier transform by
\[
\hat f(\xi) = (2\pi)^{-d/2}\int_{\mathbb{R}^d} e^{-ix\cdot \xi}f(x)\,dx.
\]
With the same normalization, the inverse Fourier transform is given by
\[
\check g(x) = (2\pi)^{-d/2}\int_{\mathbb{R}^d} e^{ix\cdot \xi}g(\xi)\,d\xi.
\]
This normalization is convenient in $L^2$, since Plancherel's theorem takes the form
\[
\|\hat f\|_{L^2(\mathbb{R}^d)} = \|f\|_{L^2(\mathbb{R}^d)}.
\]
In particular, the Fourier transform extends uniquely to a unitary operator on $L^2(\mathbb{R}^d)$.

The key point for PDE is that differentiation becomes multiplication in frequency space. Formally, if $f$ is smooth and decays sufficiently fast, then
\[
\widehat{\partial_j f}(\xi) = i\xi_j\hat f(\xi),
\]
and more generally,
\[
\widehat{\partial^{\alpha}f}(\xi) = (i\xi)^{\alpha}\hat f(\xi).
\]
Similarly, convolution becomes multiplication:
\[
\widehat{f*g}(\xi) = (2\pi)^{d/2}\hat f(\xi)\hat g(\xi).
\]
These identities explain why the Fourier transform is especially effective for linear PDE with constant coefficients.

\begin{proposition}[Fourier characterization of $H^k$]
Let $k\in \mathbb{Z}_{\ge 0}$. Then for $f\in L^2(\mathbb{R}^d)$,
\[
f\in H^k(\mathbb{R}^d)
\qquad \Longleftrightarrow \qquad
(1+|\xi|^2)^{k/2}\hat f(\xi)\in L^2(\mathbb{R}^d).
\]
Moreover,
\[
\|f\|_{H^k}^2 \simeq \int_{\mathbb{R}^d}(1+|\xi|^2)^k|\hat f(\xi)|^2\,d\xi.
\]
\end{proposition}

\begin{proof}
By Plancherel and the identity $\widehat{\partial^{\alpha}f}=(i\xi)^{\alpha}\hat f$, we have
\[
\|\partial^{\alpha}f\|_{L^2}^2 = \int_{\mathbb{R}^d}|\xi^{\alpha}|^2|\hat f(\xi)|^2\,d\xi.
\]
Summing over all $|\alpha|\le k$ yields
\[
\|f\|_{H^k}^2
= \sum_{|\alpha|\le k}\|\partial^{\alpha}f\|_{L^2}^2
\simeq \int_{\mathbb{R}^d}(1+|\xi|^2)^k|\hat f(\xi)|^2\,d\xi,
\]
since $\sum_{|\alpha|\le k}|\xi^{\alpha}|^2$ is comparable to $(1+|\xi|^2)^k$.
\end{proof}

\begin{remark}
This point of view extends naturally to fractional orders. For $s\ge 0$, one may define
\[
H^s(\mathbb{R}^d)=\Bigl\{f\in L^2(\mathbb{R}^d):(1+|\xi|^2)^{s/2}\hat f\in L^2(\mathbb{R}^d)\Bigr\},
\]
with norm
\[
\|f\|_{H^s}^2 = \int_{\mathbb{R}^d}(1+|\xi|^2)^s|\hat f(\xi)|^2\,d\xi.
\]
This also leads to negative Sobolev spaces and gives a clean frequency-space language for elliptic regularity.
\end{remark}

\begin{remark}[Poisson equation on the torus]
Suppose $f$ is defined on $\mathbb{T}^d$ and we want to solve
\[
\Delta u = f.
\]
Taking Fourier series gives
\[
-|k|^2\hat u(k)=\hat f(k),\qquad k\in \mathbb{Z}^d.
\]
Hence a necessary condition for solvability is $\hat f(0)=0$, i.e. $\int_{\mathbb{T}^d}f=0$. If this holds, then for $k\neq 0$ we may set
\[
\hat u(k) = -|k|^{-2}\hat f(k),
\]
while $\hat u(0)$ remains free, reflecting the fact that $\Delta$ has constants in its kernel. In frequency space one sees immediately that inverting $\Delta$ gains two derivatives, which is the basic elliptic regularity phenomenon.
\end{remark}


\section{Sobolev Inequalities}

\subsection{The Gagliardo--Nirenberg--Sobolev inequality}

\begin{theorem}[Gagliardo--Nirenberg--Sobolev inequality]
Let $d \ge 2$. For $u \in C_c^1(\R^d)$, we have
\[
\norm{u}_{L^{\frac{d}{d-1}}(\R^d)} \le \norm{Du}_{L^1(\R^d)}.
\]
\end{theorem}

\begin{lemma}[Loomis--Whitney inequality]
Let $d \ge 2$. For $j=1,\dots,d$, suppose
\[
f_j=f_j(x_1,\dots,\widehat{x_j},\dots,x_d).
\]
Then
\[
\left\| \prod_{j=1}^d f_j \right\|_{L^1(\R^d)}
\le
\prod_{j=1}^d \norm{f_j}_{L^{d-1}(\R^{d-1})}.
\]
\end{lemma}

\begin{remark}
The Loomis--Whitney inequality gives a geometric control of the measure of a set in $\R^d$ by the measures of its coordinate projections. In particular, if $E\subset \R^d$ and $\pi_j(E)$ denotes the $j$-th coordinate projection, then
\[
|E| \le \prod_{j=1}^d |\pi_j(E)|^{1/(d-1)}.
\]
\end{remark}

\begin{remark}
The Gagliardo--Nirenberg--Sobolev inequality is the functional counterpart of the isoperimetric inequality. Via the layer-cake decomposition, one may reduce the functional statement to geometric information on level sets.
\end{remark}

\subsection{$L^p$-based Sobolev inequalities for $1 \le p < d$}

\begin{theorem}[Sobolev inequality for $1\le p<d$]
Let $d \ge 2$ and assume $1\le p<d$. For $u\in C_c^1(\R^d)$, we have
\[
\norm{u}_{L^q(\R^d)} \le C\norm{Du}_{L^p(\R^d)},
\qquad q=\frac{dp}{d-p}.
\]
\end{theorem}

\begin{remark}
The exponent $q=\frac{dp}{d-p}$ is dictated by scaling (or dimensional analysis):
\[
[x]^{d/q}=[x]^{-1+d/p}.
\]
The condition $p<d$ is therefore unavoidable in this regime.
\end{remark}

\begin{theorem}
Let $d\ge 2$ and assume that $1\le p<d$.
\begin{enumerate}[label=\textup{(\roman*)}]
    \item For $u\in W^{1,p}(\R^d)$,
    \[
    \norm{u}_{L^q(\R^d)} \le C\norm{Du}_{L^p(\R^d)},
    \qquad q=\frac{dp}{d-p}.
    \]
    \item Let $U$ be a bounded domain. For $u\in W_0^{1,p}(U)$,
    \[
    \norm{u}_{L^q(U)} \le C\norm{Du}_{L^p(U)},
    \qquad q=\frac{dp}{d-p}.
    \]
    \item Let $U$ be a bounded domain with $C^1$ boundary. For $u\in W^{1,p}(U)$,
    \[
    \norm{u}_{L^q(U)} \le C\norm{u}_{W^{1,p}(U)},
    \qquad q=\frac{dp}{d-p}.
    \]
\end{enumerate}
\end{theorem}

\begin{remark}
In part \textup{(iii)}, one needs both $\norm{u}_{L^p}$ and $\norm{Du}_{L^p}$ in order to perform the extension argument. By contrast, in part \textup{(ii)} the vanishing trace condition removes the need for the $L^p$ term. This is why part \textup{(ii)} is often referred to as a Poincar\'{e} or Friedrichs inequality.
\end{remark}

\subsection{$L^p$-based Sobolev inequalities for $p>d$}

\begin{lemma}
Let $p>d$, $d\ge 2$, and $u\in C^\infty(\R^d)$. Then
\[
\frac{1}{|B_r(x)|}\int_{B_r(x)} |u(x)-u(y)|\,dy
\le
C\int_{B_r(x)} \frac{|Du(y)|}{|x-y|^{d-1}}\,dy.
\]
\end{lemma}

\begin{theorem}[Morrey's inequality on $\R^d$]
Let $p>d$ with $d\ge 2$, and let $u\in C^\infty(\R^d)$. Then
\[
|u(x)-u(y)| \le C|x-y|^\alpha \norm{Du}_{L^p(\R^d)},
\qquad \alpha = 1-\frac{d}{p}.
\]
\end{theorem}

\begin{remark}
The exponent
\[
\alpha = 1-\frac{d}{p}
\]
is again determined by scaling:
\[
1=\alpha + (-1)+\frac{d}{p}.
\]
In particular, Morrey's inequality is scale invariant.
\end{remark}

\begin{definition}
Let $u\in C(U)$. The H\"older seminorm of order $\alpha$ is
\[
[u]_{C^{0,\alpha}(U)} = \sup_{x\ne y} \frac{|u(x)-u(y)|}{|x-y|^\alpha}.
\]
\end{definition}

\begin{remark}
This is only a seminorm: if $[u]_{C^{0,\alpha}(U)}=0$, then $u$ is constant rather than identically zero.
\end{remark}

\begin{definition}
The H\"older norm of order $\alpha$ is
\[
\norm{u}_{C^{0,\alpha}(U)} = [u]_{C^{0,\alpha}(U)} + \norm{u}_{L^1(U)}.
\]
The corresponding H\"older space is
\[
C^{0,\alpha}(U)=\{u\in C(U): \norm{u}_{C^{0,\alpha}(U)}<\infty\}.
\]
\end{definition}

\begin{theorem}[Morrey's inequality on bounded domains]
Let $d\ge 2$, let $p>d$, and let $U\subset \R^d$ be a bounded domain with $C^1$ boundary. If $u\in W^{1,p}(U)$, then $u\in C^\alpha(U)$ with
\[
\alpha = 1-\frac{d}{p}.
\]
Moreover,
\[
\norm{u}_{C^\alpha(U)} \le C\norm{u}_{W^{1,p}(U)}.
\]
\end{theorem}

\subsection{The critical case $p=d$}

\begin{remark}
The borderline case $p=d$ does \emph{not} satisfy an estimate of the form
\[
\norm{u}_{L^\infty(U)} \le C\norm{u}_{W^{1,d}(U)}.
\]
Thus the natural target space at the endpoint is not $L^\infty$, but rather a slightly larger space.
\end{remark}

\begin{definition}
A function $u\in C^1$ has bounded mean oscillation (BMO) if
\[
\norm{u}_{\mathrm{BMO}}
=
\sup_{x\in \R^d,\,r>0}
\frac{1}{|B_r(x)|}
\int_{B_r(x)}
\left|u(y)-\frac{1}{|B_r(x)|}\int_{B_r(x)}u\,dy\right|\,dy.
\]
\end{definition}

\begin{example}
Let $U=B_1(0)\subset \R^2$ and define
\[
u(x)=\log\log\left(10+\frac{1}{|x|}\right).
\]
Then as $r=|x|\to 0$,
\[
u(r) \sim \log\log\left(\frac{1}{r}\right) \to +\infty,
\]
so $u$ is unbounded in $B_1$ and does not extend continuously to the origin.

On the other hand, one computes
\[
|\nabla u(x)| \sim \frac{1}{r\log(1/r)}
\qquad (r\to 0),
\]
and hence
\[
\int_{B_1} |\nabla u|^p \, dx < \infty
\quad \Longleftrightarrow \quad p\le 2.
\]
Therefore, in dimension $d=2$, we obtain an example with
\[
u \in W^{1,2}(B_1),
\qquad u\notin L^\infty(B_1),
\qquad u \text{ is not continuous at }0.
\]
This shows that $W^{1,d}$ does not embed into $L^\infty$ at the critical exponent.
\end{example}

\begin{theorem}[Endpoint Sobolev--BMO embedding]
Let $d\ge 2$ and let $u\in C_c^\infty(\R^d)$. Then
\[
\norm{u}_{\mathrm{BMO}} \le C(d)\norm{\nabla u}_{L^d(\R^d)}.
\]
\end{theorem}

\begin{remark}
A key tool in the proof is the Hardy--Littlewood maximal function
\[
Mu(x)=\sup_{r>0} \frac{1}{|B_r(x)|}\int_{B_r(x)} |u|,
\]
which satisfies the estimate
\[
\norm{Mu}_{L^p} \le C\norm{u}_{L^p}, \qquad 1<p\le \infty.
\]
\end{remark}

\subsection{Compactness of Sobolev embeddings: general notions}

\begin{definition}
Let $X,Y$ be normed spaces and let $T:X\to Y$ be linear. We say that $T$ is a compact operator if $T(B_X)$, the image of the unit ball in $X$, is compact in $Y$. Equivalently, for every bounded sequence $\{x_n\}\subset X$, the sequence $\{Tx_n\}$ admits a convergent subsequence in $Y$.
\end{definition}

\begin{definition}
Suppose $\iota:X\to Y$ is an embedding, i.e. a bounded linear injective map. We say that the embedding $X\hookrightarrow Y$ is compact if $\iota$ is a compact operator.
\end{definition}

\begin{remark}
For Sobolev spaces, one is interested in statements of the form
\[
W^{1,p}(U) \Subset L^q(U),
\]
where $\Subset$ denotes compact embedding. Compactness is substantially stronger than mere continuity of the embedding and is essential in many PDE arguments for obtaining strong convergence.
\end{remark}

\begin{remark}
The basic compactness principle in this direction is the Arzel\`a--Ascoli theorem. It will later be combined with regularity estimates such as Morrey's inequality.
\end{remark}

\section{Well-posedness of the FBSDE}

\subsection{Main Goal}

The main result is
\begin{theorem}
    Let $\sigma \in \Lip(\mathbb{R}^m,\mathcal{H}_+^m)$. For any $Q\in [0,\Qbar(\sigma))$, we have
    \[\Qbar(\sigma) \geq Q + \Lip(J_{\sigma}(Q,\cdot))^{-2}\] 
\end{theorem}

\begin{definition}
    $\mathcal{X}$ is the Banach space of $\mathcal{H}_+^m$-valued continuous functions on $\mathbb{R}^m$ with the norm (and this norm is finite)
    \[
    ||\sigma||_{\mathcal{X}}  := \sup_{x\in\mathbb{R}^m}\dfrac{|\sigma(x)|_F}{\langle x\rangle} 
    \]
\end{definition}

\begin{proposition}
    Prove $\mathcal{X}$ is a Banach space.
\end{proposition}

\begin{proposition}
    $\Lip(\mathbb{R}^m;\mathcal{H}_+^m) \subset \mathcal{X}$.
\end{proposition}

\subsection{SDE Solution Theory}

\begin{definition}
    Given a $\mathbb{R}^m$-valued Brownian motion $(B(q))_{q\geq 0}$ adapted to a filtration $\{\mathscr{G}_q\}_{q\geq 0}$. For an adapted process $Y$ on $[0,Q]$, a function $g:[0,Q] \times \mathbb{R}^m \to \mathcal{H}_+^m$ and a constant $a\in\mathbb{R}^m$, we define a new adapted process $\mathcal{R}^g_{a,Q}Y$ on $[0,Q]$ by
    \[\mathcal{R}^g_{a,Q}Y(q):= a + \int_0^q g(Q-p,Y(p))dB(q)\]
    whenever this stochastic integral is defined. For $Q>0$, define
    \[
    \mathcal{A}_Q:=\left\{J:[0,Q]\times \mathbb{R}^m \to \mathcal{H}_+^m\text{ continuous}: \sup_{q\in [0,Q]}\Lip(J(q,\cdot))<\infty\right\}
    \]
\end{definition}

\begin{proposition}
    Fix $L<\infty$ and $Q\in(0,\infty)$ and suppose that $g \in \mathcal{A}_Q$ satisfies \[
    \sup\limits_{q\in[0,Q]} \Lip(g(q,\cdot)) \leq L\]
    Then, for any $a\in\mathbb{R}^m$, there is a unique strong solution $\Theta^g_{a,Q}$ to the SDE
    \[
    \begin{aligned}
        d\Theta^g_{a,Q}(q) &= g(Q-q,\Theta^g_{a,Q}(q))dB(q),\quad q\in[0,Q] \\
        \Theta^g_{a,Q}(0) &= a \label{ssol}
    \end{aligned}
    \]
The solution $\Theta^g_{a,Q}$ satisfies the moment bound
    \[
    \sup_{q\in[0,Q]} \mathbb{E}|\Theta^g_{a,Q}(q)|^l<\infty\quad\text{ for all }l\in[1,\infty)
    \]
Moreover, there exists a constant $C=C(L,Q)$ such that for any $Q' \in [0,Q]$ and any adapted process
$\Gamma$ on $[0,Q']$, we have
    \[
    \sup_{q\in[0,Q']} \mathbb{E}|\Gamma(q) - \Theta^g_{a,Q}(q)|^2 \leq C \sup_{[0,Q']}\mathbb{E}|\Gamma(q) - \mathcal{R}^g_{a,Q}\Gamma(q)|^2
    \]
    and
    \[
    \mathbb{E}|\Theta^g_{a,Q}(q) - \Theta^g_{\widetilde{a},Q}(q)|^2 \leq C|a-\widetilde{a}|^2
    \]
\end{proposition}
\begin{proof}
    Define $\mathcal{K}_{L,Q'}$ on adapted processes on $[0,Q']$ by \[\sup\limits_{q\in [0,Q']} e^{-L^2q}\left(\mathbb{E}|\Gamma(q)|^2\right)^{1/2},\] which makes $\mathcal R_{a,Q}^g$ a contraction.
\end{proof}

\subsection{Local Solution}

\begin{definition}
     Let $Q>0$ and let $\sigma\in\Lip(\mathbb{R}^m;\mathcal{H}_+^m)$. We say that $J\in \mathcal{A}_Q$ is a \emph{root decoupling function} for FBSDE on $[0,Q]$ if, for all $q \in [0,Q]$ and all $a \in \mathbb{R}^m$, we have
     \[J(q,a) = [\mathbb{E}\sigma(\Theta^J_{a,q}(q))]^{1/2}\]
    In this case, we also call $J^2$ the decoupling function. In the equation above, $\Theta^J_{a,q}$ is as in $\eqref{ssol}$.
\end{definition}

\end{definition}

\begin{lemma}
    Suppose that $c\in (0,\infty), \overline{Q}<c^{-2}$ and $f:[0,\overline{Q}] \to [c^2,\infty)$ satisfies
    \[f(Q) \leq c^2\exp\left\{\int_0^Q f(q)dq\right\}\]
    for all $Q \in [0,\overline{Q}]$. Then
    \[
    f(Q) \leq (c^{-2}-Q)^{-1}\quad\text{for all }Q\in[0,\overline{Q}]
    \]
\end{lemma}
\begin{proof}
    Define $g(Q) = \int_0^Q f(q)dq$ and we have
    \[
    g'(q) = f(q),
    \]
    then
    \[
    (1 - e^{-g(q)})' = g'(q)e^{-g(q)} = f(q)e^{-g(q)} 
    \]
    and hence
    \[
    1 - e^{-g(Q)} = \int_0^Q f(q)e^{-g(q)}dq \leq c^2Q 
    \]
    Therefore, we have
    \[
    f(Q) \leq c^2e^{g(Q)} \leq \dfrac{c^2}{1-c^2Q}
    \]
    Here is another proof by \eqref{BLIneq}, consider $h$ defined on $[0,\overline{Q}]$ non-negative defined by
    \[
    h(t) = \ln(f(t)/c^2),
    \]
    then we have
    \[
    h(t) \leq \int_0^t c^2\exp(h(s))ds.
    \]
    Define
    \[
    G(x) = -e^{-x} +C
    \]
    and we have
    \[
    h(q) \leq -\ln(1-c^2q)
    \]
    for any $q \in[0,\overline{Q}]$, which means
    \[
    f(Q) \leq c^2/(1-c^2Q) = (c^{-2}-Q)^{-1}.
    \]
\end{proof}

\begin{definition}\par
    Define \[\mathcal{Q}_{\sigma}g(Q,a) = [\mathbb{E}\sigma^2(\Theta^g_{a,Q}(Q))]^{1/2},\] where $(\Theta^g_{a,Q}(q))_{q\in[0,Q]}$ from \eqref{ssol}. We note that a fixed point of $Q_{\sigma}$ is a root decoupling function for \eqref{fbsde}. We  also note that
    \[|\mathcal{Q}_{\sigma}g(Q,a)|_F^2 = |\mathbb{E}\sigma^2(\Theta^g_{a,Q}(Q))]^{1/2}|^2_F = \mathbb{E}\tr\sigma^2(\Theta^g_{a,Q}(Q)) = \mathbb{E}|\sigma(\Theta^g_{a,Q}(Q))|^2_F
    \]
    Define the set of functions
    \[
    \Lambda(M,\lambda):=\{\sigma\in \Lip(\mathbb{R}^m;\mathcal{H}_+^m):\Lip(\sigma) \leq \lambda\text{ and }|\sigma(u)|_F^2 \leq M + \lambda^2|u|^2\text{ for all }u\in\mathbb{R}^m\}
    \] For $Q_0  < \lambda^{-2}$, define the set of functions $\mathcal{Z}_{Q_0,M,\lambda}$ by
    \[
    \begin{aligned}
        \mathcal{Z}_{Q_0,M,\lambda} = \big\{g:[0,Q_0]&\times \mathbb{R}^m\to\mathcal{H}_+^m\text{ continuous}, \\
        & g(q,\cdot) \in \Lambda((1-\lambda^2 q)^{-2}M,(\lambda^{-2}-q)^{-1/2})\text{ for all }q\in [0,Q_0]\big\}
    \end{aligned}
    \]
We will construct the root decoupling function $J_{\sigma}$ as a fixed point of $Q_{\sigma}$  in a certain $\mathcal{Z}_{Q_0,M,\lambda}$.
\end{definition}

\begin{proposition}
    Fix $\lambda, M \in (0,\infty)$ and $Q_0 \in [0,\lambda^{-2})$. For any $\sigma \in \Lambda(M,\lambda)$, there is a unique root decoupling function $J_{\sigma} \in \mathcal{A}_{Q_0}$. In particular, we have
    \[
    \Qbar(\sigma) \geq \Lip(\sigma)^{-2}.
    \]
    Moreover, there is a $C = C(Q_0,M,\lambda) < \infty$ such that for any $g\in\mathcal{Z}_{Q_0,M,\lambda}$, we have
    \[
    \sup_{q\in[0,Q_0]}||(g-J_{\sigma})(q,\cdot)||_{\mathcal{X}} \leq C\sup_{q\in[0,Q_0]}||(g-\mathcal{Q}_{\sigma}g)(q,\cdot)||_{\mathcal{X}}
    \]
    and indeed
    \[
    \lim_{n\to\infty} \sup_{q\in[0,Q_0]}||(\mathcal{Q}_{\sigma}^n g - J_{\sigma})(q,\cdot)||_{\mathcal{X}} = 0
    \]
    where $Q^n_{\sigma}$ denotes the $n$-fold iterated application of $Q_{\sigma}$.
\end{proposition}
\begin{proof}
    We know any decoupling function $J\in\mathcal{A}_{Q_0}$ has to be in $\mathcal{Z}_{Q_0,M,\lambda}$ for some $M' ,\lambda \in (0,\infty)$
\end{proof}

\subsection{Extension of the Solution}

\begin{lemma}
    Let $\sigma \in \Lip(\mathbb{R}^m;\mathcal{H}^m_+)$, whenever $0\leq Q_1\leq Q_2 < \Qbar(\sigma)$, we have for any $b\in\mathbb{R}^m$,
    \[
    J_{\sigma}^2(Q_2,b) = \mathbb{E}[J_{\sigma}^2(Q_1,\Gamma^{\sigma}_{b,Q_2}(Q_2-Q_1))]
    \]
    \label{dptower}
\end{lemma}
\begin{proof}
    Remain.
\end{proof}

\begin{lemma}
    Let $\sigma \in \Lip(\mathbb{R}^m;\mathcal{H}^m_+)$, whenever $0\leq Q_1\leq Q_2 < \Qbar(\sigma)$ and if
    \[
    Q_2 - Q_1 < \Qbar(J_{\sigma}(Q_1,\cdot)),
    \]
    we have
    \[
    J_{\sigma}(Q_2,b) = J_{J_{\sigma}(Q_1,\cdot)}(Q_2-Q_1,b)\quad\text{for all }b\in\mathbb{R}^m.
    \]
\end{lemma}
\begin{proof}
    We have by $\eqref{dptower}$ that,for any $Q\in [Q_1,\Qbar(\sigma))$ and any $b\in\mathbb{R}^m$,
    \[J_{\sigma}^2(Q,b) = \mathbb{E}[J^2_{\sigma}(Q_1,\Gamma^{\sigma}_{b,Q}(Q-Q_1))]\]
    By assumption, there is a unique solution to the following FBSDE problem for $Q\in [0,Q_2-Q_1]$:
    \[
    \begin{aligned}
        d\Gamma^{J_{\sigma}(Q_1,\cdot)}_{b,Q}(q) &= J_{J_{\sigma}(Q_1,\cdot)}(Q-q,\Gamma^{J_{\sigma}(Q_1,\cdot)}_{b,Q}(q))dB(q),\quad q\in(0,Q) \\
        \Gamma^{J_{\sigma}(Q_1,\cdot)}_{b,Q}(0) &= b \\
        J_{J_{\sigma}(Q_1,\cdot)}(Q,b) &= (\mathbb{E}[J_{\sigma}^2(Q_1,\Gamma^{J_{\sigma}(Q_1,\cdot)}_{b,Q}(Q))])^{1/2}
    \end{aligned}
    \]
    Also, if given $Q\in[0,\Qbar(\sigma)-Q_1)$, $\eqref{fbsde}$ and $\eqref{dptower}$ with $Q$ replaced by $Q_1+Q$ will yield
    \[
    \begin{aligned}
        d\Gamma^{\sigma}_{b,Q+Q_1}(q) &= J_{\sigma}(Q+Q_1-q,\Gamma^{\sigma}_{b,Q+Q_1}(q))dB(q),\quad q\in(0,Q) \\
        \Gamma^{\sigma}_{b,Q+Q_1}(0) &= b \\
        J_{\sigma}(Q+Q_1,b) &= (\mathbb{E}[\sigma^2(\Gamma^{\sigma}_{b,Q+Q_1}(Q))])^{1/2} = (\mathbb{E}[J_{\sigma}^2(Q_1,\Gamma^{\sigma}_{b,Q+Q_1}(Q))])^{1/2}
    \end{aligned}
    \]
    which means $(\Gamma^{\sigma}_{b,Q+Q_1}(q)$ and $J_{\sigma}(Q+Q_1,b))$ will solve the previous FBSDE system, and by the uniqueness of FBSDE problem, we have
    \[
    J_{J_{\sigma}(Q_1,\cdot)}(Q,b) = J_{\sigma}(Q+Q_1,b)\quad\text{ for all }b\in\mathbb{R}^m\text{ and }Q\in[0,Q_2-Q_1]
    \]
\end{proof}

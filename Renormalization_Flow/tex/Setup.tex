\section{Setup}

\subsection{Semilinear SHE}

We consider the semilinear stochastic heat equation
\[du^{\rho}_t = \dfrac{1}{2}\Delta u_t^{\rho}dt + \gamma_{\rho}\sigma(u_t^{\rho})dW_t^{\rho},\quad t>0, x\in\mathbb{R}^2\label{semiSHE}\]
Here $\sigma$ is a Lipschitz nonlinearity and $dW_t^{\rho}(x)$ is a Gaussian noise that is white in time and correlated in space at scale $\rho^{1/2} << 1$. We are interested in the pointwise behavior of $u_t^{\rho}(x)$ as $\rho \to 0$, which calls for an attenuation factor $\gamma_{\rho}\sim |\ln \rho|^{-1/2}$ due to critical scaling in two dimensions. In fact, we devote most of our
attention to a variation on $\eqref{semiSHE}$ in which we first multiply $\sigma$ and then smooth the noise:
\[dv_t^{\rho} = \dfrac{1}{2}\Delta v_t^{\rho}dt + \gamma_{\rho}\htk_{\rho}[\sigma(v_t^{\rho})dW_t] \label{spde}\]

\begin{definition}\ \par
    (Space-time White Noise)\par
    Let $dW = (dW_t(x))_{t\in\mathbb{R},x\in\mathbb{R}^2}$ be a standard $\mathbb{R}^m$-valued space-time white noise generating a temporal filtration $\{\mathcal{F}_t\}_{t\in\mathbb{R}}$. Writing $dW = (dW^1,\cdots,dW^m)$ in components, then
    \[\mathbb{E}[dW_t^i(x)dW_{t'}^{i'}(x')] = \delta_{i,i'} \delta(t-t')\delta(x-x')\]
\end{definition}

\begin{definition}\ \par

The solution $v^{\rho}:\Omega\times\mathbb{R}\times\mathbb{R}^2 \to \mathbb{R}^m$ is a random vector-valued function parametrized by the correlastion parameter $\rho > 0$. \begin{definition}\ \par
    (Mild Solution 1)\par
    A mild solution for $\eqref{spde}$ is a predictable random field $v^{\rho}$ such that for all $s<t$, we have
    \[
    v^{\rho}_t(x) = \htk_{t-s}v_s^{\rho}(x) + \gamma_{\rho}\int_s^t\htk_{t+\rho-r}[\sigma(v_r^{\rho})dW_r](x)
    \]
    which means
    \[
    \begin{aligned}
        v_t^\rho{x} &= \htk_{t-s}v_s^{\rho}(x) + \gamma_{\rho}\int_s^t \int G_{t+\rho-r}(y)\sigma(v_r^{\rho})(x-y)dW_r(x-y)dy \\
        &=\htk_{t-s}v_s^{\rho}(x) + \gamma_{\rho}\int_s^t \int G_{t+\rho-r}(y)\sigma(v_r^{\rho})(x-y)dW_r(x-y)dy \\
        &=\htk_{t-s}v_s^{\rho}(x) + \gamma_{\rho}\int\left(\int_s^t G_{t+\rho-r}(y)\sigma(v_r^{\rho})(x-y)dW_r(x-y)\right)dy 
    \end{aligned}
    \]
    which can be interpreted as an Ito integral. We only look for the solution $v_t^{\rho}$ in the spaces $\mathscr{X}_t^l$ of $\mathbb{R}^m$-valued random fields $z$ on $\mathbb{R}^2$ that are $\mathcal{F}_t$-measurable and 
    \[||z||_l:=\sup_{x\in\mathbb{R}^2}(\mathbb{E}|z(x)|^l)^{1/l} < \infty\]
\end{definition}

\end{proposition}

\begin{definition}
    (Forward-backward SDE)\ \par
    The system of SDE:
    \[
    \begin{aligned}
        d\Gamma^{\sigma}_{a,Q}(q) &= J_{\sigma}(Q-q,\Gamma^{\sigma}_{a,Q}(q))dB(q),\quad &a\in\mathbb{R}^m, 0 < q <Q \\
        \Gamma^{\sigma}_{a,Q}(0) &= a, &a\in\mathbb{R}^m,Q\geq0 \\
        J_{\sigma}(q,b) &= [\mathbb{E}\sigma^2(\Gamma^{\sigma}_{a,Q}(q))]^{1/2},&q\geq 0,b\in \mathbb{R}^m
    \end{aligned}\label{fbsde}
    \]
    for $B$ a standard $\mathbb{R}^m$-valued Brownian motion and $A^{1/2}$ is the unique positive-definite matrix square root of $A\in \mathcal{H}^{m}_+$.
\end{definition}

\begin{definition}
    Given $\sigma \in \Lip(\mathbb{R}^m,\mathcal{H}_+^m)$, let $\Qbar(\sigma) \in [0,\infty]$ be the supremum of all $Q\geq 0$ such that there is a continuous function $J_{\sigma}:[0,Q]\times\mathbb{R}^m \to \mathcal{H}_+^m$ satisfyting the FBSDE and
    \[\sup_{q\in[0,\overline{Q}]} \Lip(J_{\sigma}(q,\cdot)) < \infty\]
    and define for $M,\beta \in (0,\infty)$
    \[\Sigma(M,\beta):=\{\sigma\in\Lip(\mathbb{R}^m,\mathcal{H}_+^m):|\sigma(u)|_F^2 \leq M+\beta^2|u|^2\text{ for all }u\in\mathbb{R}^m\}\]
\end{definition}

\begin{definition}\ \par
    ($L^2$-subcritical)\par
    A nonlinearity $\sigma$ is $L^2$-subcritical if $\sigma \in \Sigma(M,\beta)$ for some $M\in (0,\infty)$ and $\beta\in(0,1)$ and $\Qbar(\sigma)>1$.
\end{definition}

\subsection{Main Result}

\begin{theorem}
    Let $\sigma$ be $L^2$-subcritical. Fix $Q \in (0,1)$ and define $\tilde{\sigma}:=(1-Q)^{1/2}J_{\sigma}(Q,\cdot)$ and $\widetilde{\rho}:=\rho^{1-Q}$. Then there is a new white noise $d\widetilde{W}$ such that $\htk_{\widetilde{\rho}v}$ is an approximate mild solution of $\eqref{spde}$ with $(\widetilde{\rho},\widetilde{\sigma},d\widetilde{W})$ in place of $(\rho,\sigma,dW)$.
\end{theorem}

\begin{definition}
    Denote $\waser_2$ the Wasserstein-2 metric: for any two probability distributions $\mu,\nu$
    \[\waser_2(\mu,\nu)  = \inf_{\pi} \left(\int |x-y|^2\pi(dx,dy)\right)\]
    where $\pi$ owns marginal distributions of $\mu$ and $\nu$.\par
    Denote $\langle a\rangle := (|a|^2+1)^{1/2}$ the Japanese bracket.
\end{definition}

\begin{theorem}
    For each $L^2$-subcritical $\sigma$ and $\overline{T} \in [1,\infty)$, there is a constant $C(\sigma,\overline{T}) \in (e,\infty)$ such that for all $v_0 \in L^{\infty}(\mathbb{R}^2,\mathbb{R}^m),t\in[\overline{T}^{-1},\overline{T}]$, and $\rho \in (0,C^{âˆ’1})$, the solution $v^{\rho}$ of $\eqref{spde}$ satisfies
    \[
    \waser_2(v_t^{\rho(x)},\Gamma^{\sigma}_{a,1}(1)) \leq C\langle ||v_0||_{L^{\infty}}\rangle \sqrt{\dfrac{\ln|\ln\rho|}{|\ln\rho|}}
    \]
    where $(\Gamma^{\sigma}_{a,1}(q))_{q\in[0,1]}$ solves the FBSDE with $Q = 1$ and $a = \htk_tv_0(x)$.
\end{theorem}
